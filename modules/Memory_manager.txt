# Modules/memory_manager.py

class SessionMemory:
    def __init__(self):
        self.history = []

    def add_message(self, role: str, message: str):
        self.history.append({"role": role, "message": message})

    def get_history(self) -> list:
        return self.history

    def clear_history(self):
        self.history = []

# Singleton instance
memory = SessionMemory()
